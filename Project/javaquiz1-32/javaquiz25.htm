public class CountTest {
  static int count=0;
  int n = 0;
  public void changeCount(int chances) {
    while (n < chances) {
     n++;
     if(n%2==0)
       count++;
     }
  }
  public static void main(String[] args) {
    CountTest ct1 = new CountTest();
    CountTest ct2 = new CountTest();
    ct1.changeCount(5);

  }
}
